// Uses Puppeteer + OpenClaw browser

const { scrollFeed, likePost, commentOnPost, sendConnection } = require('./linkedin-actions');



async function runEngagementCycle() {

&nbsp; // 1. Scroll feed

&nbsp; const posts = await scrollFeed(20);

&nbsp; 

&nbsp; // 2. Identify ICP matches

&nbsp; const icpMatches = await identifyICPPosts(posts);

&nbsp; 

&nbsp; // 3. Engage

&nbsp; for (const post of icpMatches) {

&nbsp;   await likePost(post.id);

&nbsp;   

&nbsp;   // Comment on 40% of matches

&nbsp;   if (Math.random() < 0.4) {

&nbsp;     const comment = await generateComment(post);

&nbsp;     await commentOnPost(post.id, comment);

&nbsp;   }

&nbsp; }

&nbsp; 

&nbsp; // 4. Send 1-2 connection requests

&nbsp; const connectionTargets = icpMatches.slice(0, 2);

&nbsp; for (const target of connectionTargets) {

&nbsp;   const note = await generateConnectionNote(target);

&nbsp;   await sendConnection(target.profileUrl, note);

&nbsp; }

&nbsp; 

&nbsp; // 5. Log activity

&nbsp; await logActivity({

&nbsp;   timestamp: Date.now(),

&nbsp;   platform: 'LinkedIn',

&nbsp;   postsScrolled: 20,

&nbsp;   icpMatches: icpMatches.length,

&nbsp;   liked: icpMatches.length,

&nbsp;   commented: Math.floor(icpMatches.length \* 0.4),

&nbsp;   connectionsSent: connectionTargets.length

&nbsp; });

}



async function identifyICPPosts(posts) {

&nbsp; // Read ICP profile

&nbsp; const icp = await readICPProfile();

&nbsp; 

&nbsp; // Score each post

&nbsp; return posts.filter(post => {

&nbsp;   let score = 0;

&nbsp;   

&nbsp;   // Check author profile

&nbsp;   if (post.author.title.includes(icp.role)) score += 3;

&nbsp;   if (post.author.followers >= 500 \&\& post.author.followers <= 5000) score += 2;

&nbsp;   

&nbsp;   // Check content

&nbsp;   for (const painPoint of icp.painPoints) {

&nbsp;     if (post.content.includes(painPoint)) score += 2;

&nbsp;   }

&nbsp;   

&nbsp;   // Check engagement potential

&nbsp;   if (post.comments < 10 \&\& post.likes > 20) score += 1; // High engagement, low comments = opportunity

&nbsp;   

&nbsp;   return score >= 5; // Threshold for ICP match

&nbsp; });

}



async function generateComment(post) {

&nbsp; // Read voice profile

&nbsp; const voice = await readVoiceProfile('linkedin-comments');

&nbsp; 

&nbsp; // Generate value-add comment (25-40 words)

&nbsp; // Uses Claude API with voice constraints

&nbsp; const comment = await generateWithAI({

&nbsp;   prompt: `Generate a helpful comment for this post: "${post.content}". 

&nbsp;            Voice: ${voice.tone}, 

&nbsp;            Length: 25-40 words,

&nbsp;            Must add specific value, no generic praise.`,

&nbsp;   maxTokens: 100

&nbsp; });

&nbsp; 

&nbsp; return comment;

}
